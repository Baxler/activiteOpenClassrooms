# Remplace la lettre "S" par "TH"

print "Tell us something !"
user_input = gets.chomp
user_input.downcase!

if user_input.include? "s"
  user_input.gsub!(/s/, "th")
else
  print "There are no S in this word"
end

puts user_input

# Ce programme compte le nombre de fois que chaque mot apparaît dans une phrase.

puts "Entrez du texte"
text = gets.chomp
words = text.split(" ")
frequencies = Hash.new(0)
words.each { |x| 
  frequencies[x] += 1}
frequencies = frequencies.sort_by {|words, count| count}
frequencies.reverse!
frequencies.each do |x,y|
  puts "#{x} #{y}"
end

puts "Test OK"
